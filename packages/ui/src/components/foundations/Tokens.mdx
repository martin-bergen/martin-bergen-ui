# Design Tokens

Automatiskt genererad dokumentation från CSS-variabler i `index.css`.

Alla värden med `var(--...)` referenser visas med både råvärde och det upplösta värdet.

export const categories = getTokensByCategory();

export const typographyTokens = categories.flatMap((c) => 
  c.tokens.filter((t) => t.name.includes("font") || t.name.includes("text"))
);

export const otherTokens = categories.filter((c) => 
  !c.tokens.some((t) => t.name.includes("font") || t.name.includes("text"))
);

{typographyTokens.length > 0 && (
  <TypographyTable title="Typography" tokens={typographyTokens} />
)}

{otherTokens.map((category) => (
  <TokenTable key={category.name} title={category.name} tokens={category.tokens} />
))}
